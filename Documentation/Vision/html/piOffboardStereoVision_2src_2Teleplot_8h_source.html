140  void buffer(std::string const &key, std::string const& values, std::string const& flags) {
141  //Make sure buffer exists
142  if(bufferingMap_.find(key) == bufferingMap_.end()) {
143  bufferingMap_[key] = "";
144  bufferingFlushTimestampsUs_[key] = 0;
145  }
146  // Check that buffer isn't about to blow up
147  size_t keySize = key.size() + 1; // +1 is the separator
148  size_t valuesSize = bufferingMap_[key].size() + values.size() + 1; // +1 is the separator
149  size_t flagSize = 1 + flags.size(); // +1 is the separator
150  size_t nextSize = keySize + valuesSize + flagSize;
151  if(nextSize > maxBufferingSize_) {
152  flushBuffer(key, flags, true); // Force flush
153  }
154  bufferingMap_[key] += values + ";";
155  flushBuffer(key, flags);
156  }

158  void flushBuffer(std::string const& key, std::string const& flags, bool force=false) {
159  // Flush the buffer if the frequency is reached
160  int64_t nowUs = std::chrono::time_point_cast<std::chrono::microseconds>(std::chrono::system_clock::now()).time_since_epoch().count();
161  int64_t elasped = nowUs - bufferingFlushTimestampsUs_[key];
162  if(force || elasped >= static_cast<int64_t>(1e6/bufferingFrequencyHz_)) {
163  emit(formatPacket(key, bufferingMap_[key], flags));
164  bufferingMap_[key].clear();
165  bufferingFlushTimestampsUs_[key] = nowUs;
166  }
167  }
